pdf.set_font('DejaVu', size=25)

pdf.cell(200, 10, txt='Learning Python', ln=1, align='C')

pdf.cell(200, 10, txt='Welcome to the Candy shop', ln=1, align='L')

pdf.output('pdfTest.pdf')
